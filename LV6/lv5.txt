#include "mbed.h"
#include "lpc1114etf.h"

InterruptIn b1(Taster_1);
InterruptIn b2(Taster_2);
BusOut segmenti(dp18, dp4, dp28, dp6, dp5, dp27, dp26);
BusOut D(dp25, dp24, dp23);
DigitalOut tacka(dp17);
Timer t;
Ticker tiker;

double inkrement = 0;
double ukupno = 0;
void ss(){
    if (t.read_ms() < 200)
        return;
    t.reset();
    if (inkrement == 0)
        inkrement = 0.1;
    else 
        inkrement = 0;
}

void rst(){
    ukupno = 0;
    inkrement = 0;
}

void uvecaji(){
    ukupno += inkrement;   
}

int main() {
    t.start();
    int s[10] = {126, 48, 109, 121, 51, 91, 95, 112, 127, 123};
    int digits[3] = {1, 2, 4};
    D = ~3;
    segmenti = ~s[0];
    tacka = 0;
    b1.rise(&ss);
    b2.rise(&rst);
    tiker.attach(&uvecaji, 0.1);
    while (1){
        int vrijeme = ukupno * 10;
        for (int i = 1; i <= 3; i++){
            tacka = 1;
            if (i == 2)
                tacka = 0;
            D = ~digits[i - 1];
            segmenti = ~s[vrijeme % 10];
            vrijeme = vrijeme / 10;
            wait_us(10 * 1e3);
        }
    }
}

#include "mbed.h"
#include "lpc1114etf.h"

BusOut prikaz1(LED3, LED2, LED1, LED0);
BusOut prikaz2(LED7, LED6, LED5, LED4);
InterruptIn taster(Taster_1);
DigitalOut diode(LED_ACT);
Timer debounce;

Ticker t;
const float T (0.2);
int brojac1(0);
int brojac2(0);

void b1(){
    brojac1 = (brojac1 + 1) % 16;
    prikaz1 = brojac1;
}

void b2(){
    if (debounce.read_ms() < 200)
        return;
    debounce.reset();
    brojac2 = (brojac2 + 1) % 16;
    prikaz2 = brojac2;
}

int main() {
    prikaz1 = brojac1;
    prikaz2 = brojac2;
    debounce.start();
    diode = 0;
    t.attach(&b1, T);
    taster.rise(&b2);
    while(1);
}
